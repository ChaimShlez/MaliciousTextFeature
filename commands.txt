docker network create mongo-kafka-project


docker run -d \
  --name mongodb-project \
  --network mongo-kafka-project \
  -e MONGO_INITDB_ROOT_USERNAME=admin \
  -e MONGO_INITDB_ROOT_PASSWORD=secretpassword \
  -e MONGO_INITDB_DATABASE=mydb \
  -p 27020:27017 \
  mongo:7.0.23




 docker run -d `
  --name kafka-project `
  --network mongo-kafka-project `
  -p 9095:9092 `
  -e ALLOW_PLAINTEXT_LISTENER=yes `
  -e KAFKA_BROKER_ID=1 `
  -e KAFKA_CFG_NODE_ID=1 `
  -e KAFKA_CFG_PROCESS_ROLES=broker,controller `
  -e KAFKA_CFG_LISTENERS=PLAINTEXT://:9092,CONTROLLER://:9095 `
  -e KAFKA_CFG_ADVERTISED_LISTENERS=PLAINTEXT://kafka-project:9092 `
  -e KAFKA_CFG_CONTROLLER_LISTENER_NAMES=CONTROLLER `
  -e KAFKA_CFG_CONTROLLER_QUORUM_VOTERS=1@kafka-project:9095 `
  -e KAFKA_CFG_LISTENER_SECURITY_PROTOCOL_MAP=PLAINTEXT:PLAINTEXT,CONTROLLER:PLAINTEXT `
  bitnami/kafka:3.5
